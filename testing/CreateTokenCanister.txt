
The below file contains the DFX commands to create the $FPL token canister:



//Set the script variables:
export NETWORK=local
export MINTER_PRINCIPAL=$(dfx identity get-principal)
export ARCHIVE_CONTROLLER=$(dfx identity get-principal)
export TOKEN_NAME="FPL"
export TOKEN_SYMBOL=FPL


//Sciprt to create token
dfx deploy --network ${NETWORK} token_canister --mode=reinstall --argument '(variant { Init = 
     record {
       token_name = "'${TOKEN_NAME}'";
       token_symbol = "'${TOKEN_SYMBOL}'";
       minting_account = record { owner = principal "'${MINTER_PRINCIPAL}'";};
       initial_balances = vec {                                
            record {                                            
                record {                                        
                    owner = principal "'${MINTER_PRINCIPAL}'";   
                    subaccount = null;                          
                };                                              
                100_000_000                                 
            }                                                   
        };  
       metadata = vec {};
       transfer_fee = 1;
       archive_options = record {
         trigger_threshold = 2000;
         num_blocks_to_archive = 1000;
         controller_id = principal "'${ARCHIVE_CONTROLLER}'";
       }
}})'
